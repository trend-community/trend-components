```markup
  <!-- Primary button -->
  <button class="tc-Button" type="button">
    button
  </button>

  <!-- Accent Button -->
   <button class="tc-Button tc-Button--accent" type="button">
    accent
  </button>

  <!-- Button with a child icon -->
  <button class="tc-Button" type="button">
    <svg class="tc-Button-icon">...</svg>
    icon
  </button>

  <!-- Compact button -->
  <button class="tc-Button tc-Button--compact" type="button">
    compact
  </button>

  <!-- Disabled button -->
  <button class="tc-Button [is-disabled]" disabled type="button">
    disabled
  </button>
```
